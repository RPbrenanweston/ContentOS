   ```javascript
   const axios = require('axios');
   const pRetry = require('p-retry');
   const config = require('../config');

   const attioClient = axios.create({
     baseURL: 'https://api.attio.com/v2',
     headers: {
       'Authorization': `Bearer ${process.env.ATTIO_API_KEY}`,
       'Content-Type': 'application/json'
     },
     timeout: 30000
   });

   async function syncAccountNote(accountId, noteText, metadata = {}) {
     return pRetry(
       async () => {
         const response = await attioClient.post(`/notes`, {
           data: {
             parent_object: 'companies',
             parent_record_id: accountId,
             title: `Daily Digest - ${new Date().toISOString().split('T')[0]}`,
             content_plaintext: noteText,
             ...metadata
           }
         });

         console.log(`[Attio] Note synced for account ${accountId}`);
         return response.data;
       },
       {
         onFailedAttempt: error => {
           const status = error.response?.status;
           const retryAfter = error.response?.headers?.['retry-after'];

           if (status === 429) {
             console.log(`[Attio] Rate limited (429). Retry-After: ${retryAfter || 'not specified'}s`);
             // pRetry will handle backoff
           } else if (status >= 500) {
             console.log(`[Attio] Server error (${status}). Retrying...`);
           } else {
             // Don't retry client errors (400, 401, 403, 404)
             throw error;
           }
         },
         retries: 5,
         minTimeout: 1000,
         maxTimeout: 30000,
         factor: 2 // Exponential: 1s, 2s, 4s, 8s, 16s
       }
     );
   }

   async function enrichProspect(prospectId, trendData) {
     return pRetry(
       async () => {
         const response = await attioClient.patch(`/objects/people/${prospectId}`, {
           data: {
             attributes: {
               hiring_trends: trendData.trends?.join(', ') || '',
               last_signal_date: trendData.lastSignalDate || new Date().toISOString(),
               signal_count: trendData.signalCount || 0
             }
           }
         });

         console.log(`[Attio] Prospect ${prospectId} enriched`);
         return response.data;
       },
       {
         onFailedAttempt: error => {
           const status = error.response?.status;
           if (status === 429 || status >= 500) {
             console.log(`[Attio] Retry for prospect ${prospectId}: ${status}`);
           } else {
             throw error;
           }
         },
         retries: 3,
         minTimeout: 1000,
         maxTimeout: 10000
       }
     );
   }

   async function testConnection() {
     try {
       const response = await attioClient.get('/workspace');
       console.log(`[Attio] Connected to workspace: ${response.data?.data?.name || 'Unknown'}`);
       return true;
     } catch (error) {
       console.error(`[Attio] Connection test failed: ${error.message}`);
       return false;
     }
   }

   module.exports = { attioClient, syncAccountNote, enrichProspect, testConnection };
   ```

1. Create src/schedules/attio-sync.js:
   ```javascript
   const schedule = require('node-schedule');
   const { createClient } = require('@supabase/supabase-js');
   const config = require('../config');
   const { syncAccountNote, enrichProspect } = require('../integrations/attio');
   const { formatAsNote } = require('../reporters/formatter');

   const supabase = createClient(config.supabase.url, config.supabase.serviceKey);

   async function getDigestsToSync() {
     // Get today's digests that haven't been synced to Attio yet
     const today = new Date().toISOString().split('T')[0];

     const { data, error } = await supabase
       .from('digest_history')
       .select('*, companies(*)')
       .gte('generated_at', `${today}T00:00:00Z`)
       .is('attio_synced_at', null);

     if (error) {
       console.error(`[AttioSync] Failed to fetch digests: ${error.message}`);
       return [];
     }
     return data || [];
   }

   async function syncAllToAttio() {
     console.log('[AttioSync] Starting daily sync');

     const digests = await getDigestsToSync();
     console.log(`[AttioSync] Found ${digests.length} digests to sync`);

     const results = { success: 0, failed: 0, errors: [] };

     // Process with controlled concurrency (5 at a time to stay under rate limit)
     const BATCH_SIZE = 5;
     for (let i = 0; i < digests.length; i += BATCH_SIZE) {
       const batch = digests.slice(i, i + BATCH_SIZE);

       await Promise.all(batch.map(async (digest) => {
         const accountId = digest.account_id;
         try {
           // Format digest as Attio note
           const noteText = formatAsNote(
             digest.companies,
             digest.signals || [],
             digest.trends || [],
             digest.relevance_copy || []
           );

           // Sync account note
           await syncAccountNote(accountId, noteText, {
             source: 'digest_reporter',
             digest_id: digest.id
           });

           // Mark as synced in Supabase
           await supabase
             .from('digest_history')
             .update({ attio_synced_at: new Date().toISOString() })
             .eq('id', digest.id);

           results.success++;
           console.log(`[AttioSync] Synced account ${accountId}`);

         } catch (error) {
           results.failed++;
           results.errors.push({ accountId, error: error.message });
           console.error(`[AttioSync] Failed for ${accountId}: ${error.message}`);

           // Log failure for manual backfill
           await supabase
             .from('sync_failures')
             .insert({
               account_id: accountId,
               sync_type: 'attio',
               error_message: error.message,
               created_at: new Date().toISOString()
             });
         }
       }));

       // Small delay between batches to be safe on rate limits
       if (i + BATCH_SIZE < digests.length) {
         await new Promise(r => setTimeout(r, 200));
       }
     }

     console.log(`[AttioSync] Completed: ${results.success} success, ${results.failed} failed`);
     return results;
   }

   function scheduleAttioSync() {
     // Run 30 minutes after digest generation (9:30 AM in each timezone)
     // This ensures digests are generated before sync attempts
     schedule.scheduleJob('30 9 * * *', async () => {
       console.log('[AttioSync] Scheduled sync triggered');
       await syncAllToAttio();
     });

     console.log('[AttioSync] Daily sync scheduled for 9:30 AM');
   }

   module.exports = { syncAllToAttio, scheduleAttioSync, getDigestsToSync };
   ```
