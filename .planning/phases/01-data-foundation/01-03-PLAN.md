# Plan 01-03: RSS Feed Aggregator

## Goal

Build an RSS feed aggregator that polls tech news, security industry sources, and LinkedIn company posts every 24 hours, extracts signals (HIRING, COMPANY, INDIVIDUAL), classifies them with confidence scores, deduplicates intelligently, and stores them in the Supabase signals table linked to companies. This provides zero-cost automated signal detection for the 10 pilot accounts.

## Must-haves

1. **RSS feed sources configured** for 3 signal categories:
   - Tech news (TechCrunch, CRN, VentureBeat) for company growth signals
   - Security industry (Bleeping Computer, SecurityWeek, Dark Reading) for sector-specific hiring
   - LinkedIn company posts (if feasible via RSS) for direct company announcements
2. **Feed parser** that extracts title, link, date, source, and AI-extracted summary from feed items
3. **Signal classifier** that categorizes each item as HIRING, COMPANY, or INDIVIDUAL with flexible tags (hiring_urgency, competitor_signal, expansion_signal)
4. **Company matching engine** that links signals to companies in database via name matching (exact/fuzzy) AND keyword matching
5. **Confidence scoring algorithm** (0-100) based on source type, pattern match quality, and company relevance
6. **Deduplication strategy** (hybrid: one signal record per opportunity, track all sources that mentioned it)
7. **Daily polling scheduler** (24-hour cadence, aligns with sales team workflow)
8. **Storage in Supabase** signals table with proper company linkage and metadata
9. **Relevance flagging** to mark obviously irrelevant signals (store all, but flag for filtering)
10. **Documentation** of feed sources, classification logic, and confidence scoring rules

```python
import feedparser
import requests
from bs4 import BeautifulSoup
from datetime import datetime
import yaml
from pathlib import Path
import logging

class RSSAggregator:
    def __init__(self, config_path='backend/feed_config.yaml'):
        with open(config_path) as f:
            self.config = yaml.safe_load(f)
        self.logger = logging.getLogger(__name__)

    def fetch_all_feeds(self):
        """Fetch all RSS feeds from config"""
        all_items = []

        for category, feeds in self.config['feeds'].items():
            for feed_config in feeds:
                try:
                    items = self.fetch_feed(feed_config)
                    all_items.extend(items)
                    self.logger.info(f"Fetched {len(items)} items from {feed_config['name']}")
                except Exception as e:
                    self.logger.error(f"Failed to fetch {feed_config['name']}: {e}")

        return all_items

    def fetch_feed(self, feed_config):
        """Fetch single RSS feed"""
        feed = feedparser.parse(feed_config['url'])
        items = []

        for entry in feed.entries:
            item = {
                'title': entry.get('title', ''),
                'link': entry.get('link', ''),
                'published': self._parse_date(entry.get('published', '')),
                'summary': entry.get('summary', ''),
                'source': feed_config['name'],
                'source_category': feed_config['category'],
                'confidence_boost': feed_config.get('confidence_boost', 0),
                'raw_content': self._fetch_full_content(entry.get('link', ''))
            }
            items.append(item)

        return items

    def _parse_date(self, date_string):
        """Parse RSS date to datetime"""
        from dateutil import parser
        try:
            return parser.parse(date_string)
        except:
            return datetime.now()

    def _fetch_full_content(self, url):
        """Fetch full article content (optional, for AI summarization)"""
```

```python
import re
from typing import List, Dict

class SignalClassifier:
    def __init__(self, keywords_config):
        self.keywords = keywords_config

    def classify(self, item: Dict) -> Dict:
        """
        Classify feed item into signal type and extract tags

        Returns:
        {
            'signal_type': 'HIRING' | 'COMPANY' | 'INDIVIDUAL',
            'tags': ['hiring_urgency:high', 'expansion_signal', ...],
            'matched_keywords': [...],
            'pattern_confidence': 0-100
        }
        """
        text = f"{item['title']} {item['summary']}".lower()

        # Detect signal type
        signal_type = self._detect_signal_type(text)

        # Extract tags
        tags = self._extract_tags(text)

        # Match keywords
        matched_keywords = self._match_keywords(text)

        # Calculate pattern confidence
        pattern_confidence = self._calculate_pattern_confidence(
            signal_type, matched_keywords, tags
        )

        return {
            'signal_type': signal_type,
            'tags': tags,
            'matched_keywords': matched_keywords,
            'pattern_confidence': pattern_confidence
        }

    def _detect_signal_type(self, text: str) -> str:
        """Detect primary signal type"""
        hiring_score = self._count_keyword_matches(text, self.keywords['hiring_signals'])
        leadership_score = self._count_keyword_matches(text, self.keywords['leadership_signals'])
        company_score = (
            self._count_keyword_matches(text, self.keywords['funding_signals']) +
            self._count_keyword_matches(text, self.keywords['product_signals']) +
            self._count_keyword_matches(text, self.keywords['expansion_signals'])
        )

        # Classify by highest score
        if hiring_score >= leadership_score and hiring_score >= company_score:
            return 'HIRING'
        elif leadership_score > company_score:
            return 'INDIVIDUAL'
        else:
            return 'COMPANY'

    def _extract_tags(self, text: str) -> List[str]:
        """Extract flexible tags based on content"""
        tags = []

        # Hiring urgency
        if any(kw in text for kw in ['urgently', 'immediately', 'asap', 'rapidly growing']):
            tags.append('hiring_urgency:high')
        elif any(kw in text for kw in ['hiring', 'recruiting', 'positions']):
            tags.append('hiring_urgency:medium')

        # Expansion signals
        if any(kw in text for kw in self.keywords['expansion_signals']):
            tags.append('expansion_signal')

        # Funding signals
        if any(kw in text for kw in self.keywords['funding_signals']):
            tags.append('funding_signal')

        # Leadership changes
        if any(kw in text for kw in self.keywords['leadership_signals']):
            tags.append('leadership_change')

        # Competitor signals (if competitor names mentioned)
        # Placeholder: would require competitor database
        # if self._mentions_competitor(text):
        #     tags.append('competitor_signal')

        return tags

    def _match_keywords(self, text: str) -> List[str]:
        """Return list of matched keywords"""
        matched = []
        for category, keywords in self.keywords.items():
            for keyword in keywords:
                if keyword in text:
                    matched.append(keyword)
        return matched

    def _calculate_pattern_confidence(self, signal_type, matched_keywords, tags) -> int:
        """Calculate 0-100 confidence based on pattern matching"""
        confidence = 50  # Base confidence

        # Boost for keyword matches
        confidence += min(len(matched_keywords) * 5, 30)

        # Boost for relevant tags
        confidence += min(len(tags) * 5, 20)

        # Cap at 100
        return min(confidence, 100)

    def _count_keyword_matches(self, text: str, keywords: List[str]) -> int:
        """Count how many keywords from list appear in text"""
        return sum(1 for kw in keywords if kw in text)
```

### 5. Build Company Matching Engine

**Create `backend/company_matcher.py`:**

```python
from fuzzywuzzy import fuzz
from typing import List, Dict, Optional

class CompanyMatcher:
    def __init__(self, supabase_client):
        self.client = supabase_client
        self.companies = self._load_companies()

    def _load_companies(self) -> List[Dict]:
        """Load all companies from database into memory (for faster matching)"""
        result = self.client.table('companies').select('id, name, domain').execute()
        return result.data

    def match_signal_to_company(self, item: Dict) -> Optional[Dict]:
        """
        Match feed item to company in database

        Strategy:
        1. Exact name match (highest confidence)
        2. Fuzzy name match (medium confidence)
        3. Domain extraction from text (high confidence if found)
        4. Keyword match only (low confidence, flag for review)

        Returns:
        {
            'company_id': uuid or None,
            'match_type': 'exact' | 'fuzzy' | 'domain' | 'keyword' | 'none',
            'match_confidence': 0-100,
            'matched_company_name': str
        }
        """
        text = f"{item['title']} {item['summary']}"

        # Try exact name match
        for company in self.companies:
            if company['name'].lower() in text.lower():
                return {
                    'company_id': company['id'],
                    'match_type': 'exact',
                    'match_confidence': 100,
                    'matched_company_name': company['name']
                }

        # Try fuzzy name match
        for company in self.companies:
            ratio = fuzz.partial_ratio(company['name'].lower(), text.lower())
            if ratio > 85:  # Threshold for fuzzy match
                return {
                    'company_id': company['id'],
                    'match_type': 'fuzzy',
                    'match_confidence': ratio,
                    'matched_company_name': company['name']
                }

        # Try domain extraction
        domains = self._extract_domains(text)
        for domain in domains:
            for company in self.companies:
                if company['domain'] == domain:
                    return {
                        'company_id': company['id'],
                        'match_type': 'domain',
                        'match_confidence': 95,
                        'matched_company_name': company['name']
                    }

        # No match
        return {
            'company_id': None,
            'match_type': 'none',
            'match_confidence': 0,
            'matched_company_name': None
        }

    def _extract_domains(self, text: str) -> List[str]:
        """Extract domain names from text"""
        import re
        # Simple regex for domains
        pattern = r'\b(?:[a-z0-9](?:[a-z0-9-]{0,61}[a-z0-9])?\.)+[a-z0-9][a-z0-9-]{0,61}[a-z0-9]\b'
        domains = re.findall(pattern, text.lower())
        return domains
```

### 6. Build Confidence Scorer

**Create `backend/confidence_scorer.py`:**

```python
class ConfidenceScorer:
    def calculate_final_confidence(
        self,
        source_confidence_boost: int,
        pattern_confidence: int,
        match_confidence: int,
        match_type: str
    ) -> int:
        """
        Calculate final 0-100 confidence score

        Factors:
        1. Source type (feed confidence boost from config)
        2. Pattern match quality (from classifier)
        3. Company match quality (from matcher)
        4. Match type (exact > domain > fuzzy > keyword)

        Formula:
        base = 30
        + source_boost (0-20 from config)
        + pattern_confidence * 0.3 (0-30)
        + match_confidence * 0.2 (0-20)
        + match_type_boost (0-30)
        """
        base = 30

        # Source boost (from feed config)
        source_score = min(source_confidence_boost, 20)

        # Pattern score (30% weight)
        pattern_score = int(pattern_confidence * 0.3)

        # Match score (20% weight)
        match_score = int(match_confidence * 0.2)

        # Match type boost
        match_type_boost = {
            'exact': 30,
            'domain': 25,
            'fuzzy': 15,
            'keyword': 5,
            'none': 0
        }.get(match_type, 0)

        final_score = base + source_score + pattern_score + match_score + match_type_boost

        # Cap at 100
        return min(final_score, 100)
```

**Create `backend/deduplicator.py`:**

```python
from datetime import datetime, timedelta

class SignalDeduplicator:
    def __init__(self, supabase_client):
        self.client = supabase_client

    def is_duplicate(self, new_signal: Dict) -> Dict:
        """
        Check if signal already exists in database

        Deduplication strategy:
        - One signal record per opportunity
        - Track all sources that mentioned it (in metadata)
        - Match criteria: same company + similar title + within 7 days

        Returns:
        {
            'is_duplicate': bool,
            'existing_signal_id': uuid or None,
            'should_merge': bool
        }
        """
        # Query for similar signals
        seven_days_ago = (datetime.now() - timedelta(days=7)).isoformat()

        existing = self.client.table('signals').select('*').match({
            'company_id': new_signal['company_id'],
            'signal_type': new_signal['signal_type']
        }).gte('detected_at', seven_days_ago).execute()

        # Check for title similarity
        for signal in existing.data:
            similarity = self._calculate_title_similarity(
                signal['title'],
                new_signal['title']
            )

            if similarity > 0.8:  # 80% similar
                return {
                    'is_duplicate': True,
                    'existing_signal_id': signal['id'],
                    'should_merge': True
                }

        return {
            'is_duplicate': False,
            'existing_signal_id': None,
            'should_merge': False
        }

    def merge_signal_sources(self, existing_signal_id: str, new_source: Dict):
        """
        Merge new source into existing signal metadata

        Tracks: which sources mentioned this signal, when
        """
        # Get existing signal
        signal = self.client.table('signals').select('*').eq('id', existing_signal_id).execute().data[0]

        # Update metadata to track sources
        if 'sources' not in signal.get('metadata', {}):
            signal['metadata'] = {'sources': []}

        signal['metadata']['sources'].append({
            'source': new_source['source'],
            'url': new_source['link'],
            'detected_at': datetime.now().isoformat()
        })

        # Update signal
        self.client.table('signals').update({
            'metadata': signal['metadata'],
            'updated_at': datetime.now().isoformat()
        }).eq('id', existing_signal_id).execute()

    def _calculate_title_similarity(self, title1: str, title2: str) -> float:
        """Calculate similarity between two titles (0-1)"""
        from fuzzywuzzy import fuzz
        return fuzz.ratio(title1.lower(), title2.lower()) / 100.0
```

### 8. Build Signal Pipeline Orchestrator

**Update `backend/rss_aggregator.py` with pipeline:**

```python
class SignalPipeline:
    def __init__(self, supabase_client, config_path='backend/feed_config.yaml'):
        self.aggregator = RSSAggregator(config_path)
        self.classifier = SignalClassifier(self.aggregator.config['keywords'])
        self.matcher = CompanyMatcher(supabase_client)
        self.scorer = ConfidenceScorer()
        self.deduplicator = SignalDeduplicator(supabase_client)
        self.client = supabase_client
        self.logger = logging.getLogger(__name__)

    def run_pipeline(self):
        """
        Full RSS → Signals pipeline:
        1. Fetch all RSS feeds
        2. Classify each item
        3. Match to companies
        4. Score confidence
        5. Deduplicate
        6. Store in Supabase
        """
        # Fetch feeds
        items = self.aggregator.fetch_all_feeds()
        self.logger.info(f"Fetched {len(items)} feed items")

        processed_count = 0
        duplicate_count = 0
        unmatched_count = 0

        for item in items:
            try:
                # Classify
                classification = self.classifier.classify(item)

                # Match to company
                match_result = self.matcher.match_signal_to_company(item)

                # Skip if no company match (store for future review)
                if match_result['company_id'] is None:
                    unmatched_count += 1
                    self._store_unmatched_signal(item, classification)
                    continue

                # Score confidence
                final_confidence = self.scorer.calculate_final_confidence(
                    source_confidence_boost=item['confidence_boost'],
                    pattern_confidence=classification['pattern_confidence'],
                    match_confidence=match_result['match_confidence'],
                    match_type=match_result['match_type']
                )

                # Check for duplicates
                signal_data = {
                    'company_id': match_result['company_id'],
                    'signal_type': classification['signal_type'],
                    'title': item['title'],
                    'description': item['summary'],
                    'source': item['source'],
                    'source_url': item['link'],
                    'tags': classification['tags'],
                    'confidence_score': final_confidence,
                    'detected_at': item['published'],
                    'is_active': True,
                    'metadata': {
                        'matched_keywords': classification['matched_keywords'],
                        'match_type': match_result['match_type'],
                        'source_category': item['source_category']
                    }
                }

                dedup_result = self.deduplicator.is_duplicate(signal_data)

                if dedup_result['is_duplicate']:
                    # Merge sources
                    self.deduplicator.merge_signal_sources(
                        dedup_result['existing_signal_id'],
                        {'source': item['source'], 'link': item['link']}
                    )
                    duplicate_count += 1
                else:
                    # Insert new signal
                    self.client.table('signals').insert(signal_data).execute()
                    processed_count += 1

            except Exception as e:
                self.logger.error(f"Failed to process item: {e}")

        return {
            'processed': processed_count,
            'duplicates': duplicate_count,
            'unmatched': unmatched_count,
            'timestamp': datetime.now().isoformat()
        }

    def _store_unmatched_signal(self, item, classification):
        """Store unmatched signals for future review"""
        # Store in separate table or with flag for manual review
        pass
```
