# Plan 03-03: Dashboard + Measurement

## Goal

Build feedback loop dashboard to visualize agent performance, measure improvement over time, and validate that the learning loop achieves ≥20% improvement in signal quality after 100+ feedback entries.

## Must-haves

1. **Feedback Loop Dashboard**
   - Real-time thumbs-up rate by agent (researcher, resourcer, copywriter)
   - Agent performance trends over time (approval rate by week)
   - Top-performing prompt versions (highest approval rates)
   - Recent learning insights display
   - Feedback volume metrics (total feedback count, feedback per agent)

2. **Performance Measurement Module**
   - Calculate baseline approval rate (first 50 feedback entries)
   - Calculate current approval rate (most recent 50 feedback entries)
   - Track improvement percentage: `(current_rate - baseline_rate) / baseline_rate * 100`
   - Verify ≥20% improvement threshold
   - Statistical significance test (ensure improvement not random)

3. **A/B Testing Metrics**
   - Compare prompt version A vs B side-by-side
   - Show approval rate, sample size, confidence interval
   - Declare winner when statistical significance reached
   - Auto-deactivate losing version

4. **Visualization Components**
   - Line chart: approval rate over time (weekly data points)
   - Bar chart: feedback count by agent
   - Comparison table: prompt version performance
   - Trend indicators: ↑ improving, ↓ declining, → stable

5. **End-to-End Validation**
   - Test with 100+ sample feedback entries
   - Verify learning loop execution
   - Measure actual improvement percentage
   - Generate validation report

export async function calculateImprovement(
  agentType: AgentType
): Promise<ImprovementReport> {
  // 1. Get first 50 feedback entries (baseline)
  const baselineFeedback = await getFirstNFeedback(agentType, 50);
  const baselineRate = calculateApprovalRate(baselineFeedback);

  // 2. Get most recent 50 feedback entries (current)
  const currentFeedback = await getRecentNFeedback(agentType, 50);
  const currentRate = calculateApprovalRate(currentFeedback);

  // 3. Calculate improvement
  const improvementPercentage = ((currentRate - baselineRate) / baselineRate) * 100;
  const improvementAbsolute = currentRate - baselineRate;

  // 4. Statistical significance (two-proportion z-test)
  const pValue = calculateSignificance(baselineFeedback, currentFeedback);

  // 5. Generate conclusion
  const meetsThreshold = improvementPercentage >= 20;
  const isSignificant = pValue < 0.05;
  const conclusion = generateConclusion(meetsThreshold, isSignificant, improvementPercentage);

  return {
    agent_type: agentType,
    baseline_rate: baselineRate,
    current_rate: currentRate,
    improvement_percentage: improvementPercentage,
    improvement_absolute: improvementAbsolute,
    meets_threshold: meetsThreshold,
    baseline_sample_size: baselineFeedback.length,
    current_sample_size: currentFeedback.length,

    conclusion
  };

  } else if (meetsThreshold && !isSignificant) {

// GET /api/analytics/feedback?agent_type={type}
export async function GET(request: Request) {
  const { searchParams } = new URL(request.url);
  const agentType = searchParams.get('agent_type') as AgentType;
  const action = searchParams.get('action'); // 'metrics' | 'improvement' | 'versions'

  if (action === 'improvement') {
    const report = await calculateImprovement(agentType);
    return Response.json(report);
  } else if (action === 'versions') {
    const versions = await getPromptVersionComparison(agentType);
    return Response.json(versions);
  } else {
    const metrics = await getFeedbackMetrics(agentType);
    return Response.json(metrics);
  }
}

async function validateLearningLoop() {
  console.log('=== Learning Loop Validation ===\n');

  // Step 1: Check feedback data volume
  const totalFeedback = await getTotalFeedbackCount();
  console.log(`Total feedback entries: ${totalFeedback}`);

  if (totalFeedback < 100) {
    console.error('❌ Insufficient feedback data (need ≥100 entries)');
    return;
  }

  // Step 2: Validate for each agent
  for (const agentType of ['researcher', 'resourcer', 'copywriter'] as AgentType[]) {
    console.log(`\n--- ${agentType.toUpperCase()} ---`);

    // Get improvement report
    const report = await calculateImprovement(agentType);
    console.log(`Baseline rate: ${(report.baseline_rate * 100).toFixed(1)}%`);
    console.log(`Current rate: ${(report.current_rate * 100).toFixed(1)}%`);
    console.log(`Improvement: ${report.improvement_percentage.toFixed(1)}%`);
    console.log(`Meets ≥20% threshold: ${report.meets_threshold ? '✓' : '✗'}`);
    console.log(`Statistical significance (p-value): ${report.statistical_significance.toFixed(4)}`);
    console.log(`Conclusion: ${report.conclusion}`);

    // Check learning insights exist
    const insights = await getRecentLearningInsights(agentType, 1);
    console.log(`Recent learning insights: ${insights.length > 0 ? '✓' : '✗'}`);

    // Check prompt versions
    const versions = await getPromptVersionHistory(agentType);
    console.log(`Prompt versions created: ${versions.length}`);
    console.log(`Active version: ${versions.find(v => v.is_active)?.version}`);
  }

  // Step 3: Overall validation
  console.log('\n=== VALIDATION SUMMARY ===');
  const allReports = await Promise.all(
    ['researcher', 'resourcer', 'copywriter'].map(t => calculateImprovement(t as AgentType))
  );

  const allMeetThreshold = allReports.every(r => r.meets_threshold);
  const avgImprovement = allReports.reduce((sum, r) => sum + r.improvement_percentage, 0) / 3;

  console.log(`All agents meet ≥20% improvement: ${allMeetThreshold ? '✓' : '✗'}`);
  console.log(`Average improvement: ${avgImprovement.toFixed(1)}%`);

  if (allMeetThreshold) {
    console.log('\n✓✓✓ LEARNING LOOP VALIDATION PASSED ✓✓✓');
  } else {
    console.log('\n⚠⚠⚠ LEARNING LOOP VALIDATION INCOMPLETE ⚠⚠⚠');
    console.log('Some agents have not yet reached 20% improvement threshold.');
    console.log('Continue collecting feedback and re-run validation.');
  }
}
