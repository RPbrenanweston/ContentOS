3. Create src/integrations/slack.js:
   ```javascript
   const { WebClient } = require('@slack/web-api');
   const config = require('../config');

   const slackClient = new WebClient(config.slack.botToken);

   async function sendDigest(channelId, accountId, blocks) {
     try {
       const result = await slackClient.chat.postMessage({
         channel: channelId || config.slack.channelId,
         blocks: blocks,
         text: `Daily digest for account ${accountId}`, // Fallback
         metadata: {
           event_type: 'digest_sent',
           event_payload: {
             account_id: accountId,
             timestamp: new Date().toISOString()
           }
         }
       });

       console.log(`[Slack] Digest posted: ts=${result.ts}, account=${accountId}`);
       return { success: true, ts: result.ts, channel: result.channel };
     } catch (error) {
       console.error(`[Slack] Failed to post digest: ${error.message}`);
       // Don't throw - let caller handle async failure
       return { success: false, error: error.message };
     }
   }

   module.exports = { slackClient, sendDigest };
   ```

4. Create src/integrations/feedback-handler.js:
   ```javascript
   const { createClient } = require('@supabase/supabase-js');
   const config = require('../config');

   const supabase = createClient(config.supabase.url, config.supabase.serviceKey);

   async function handleFeedbackInteraction(payload) {
     const { actions, user, message } = payload;

     for (const action of actions) {
       if (action.type === 'button' && action.action_id.startsWith('feedback_')) {
         // Parse action_id: feedback_useful_123 or feedback_notuseful_123
         const parts = action.action_id.split('_');
         const feedbackType = parts[1]; // 'useful' or 'notuseful'
         const accountId = parts[2];

         // Store feedback in Supabase
         const { data, error } = await supabase
           .from('digest_feedback')
           .insert({
             account_id: accountId,
             feedback_type: feedbackType,
             user_id: user.id,
             user_name: user.username || user.name,
             message_ts: message?.ts,
             created_at: new Date().toISOString()
           });

         if (error) {
           console.error(`[Feedback] Failed to store: ${error.message}`);
           return { response_type: 'ephemeral', text: 'Failed to record feedback. Please try again.' };
         }

         console.log(`[Feedback] Stored: account=${accountId}, type=${feedbackType}, user=${user.id}`);
         return {
           response_type: 'ephemeral',
           text: `Feedback recorded. Thank you!`
         };
       }
     }

     return { response_type: 'ephemeral', text: 'Unknown action.' };
   }

   module.exports = { handleFeedbackInteraction };
   ```

1. Create src/schedules/digest-scheduler.js:
   ```javascript
   const schedule = require('node-schedule');
   const { DateTime } = require('luxon');
   const { createClient } = require('@supabase/supabase-js');
   const config = require('../config');
   const { generateDigest } = require('../reporters');
   const { sendDigest } = require('../integrations/slack');

   const supabase = createClient(config.supabase.url, config.supabase.serviceKey);

   // Track last run to prevent duplicate sends
   const lastRunByTimezone = {};

   async function getActiveAccounts() {
     const { data, error } = await supabase
       .from('companies')
       .select('id, name')
       .eq('active', true)
       .limit(10); // Pilot: 10 accounts

     if (error) {
       console.error(`[Scheduler] Failed to fetch accounts: ${error.message}`);
       return [];
     }
     return data || [];
   }

   async function runDigestForTimezone(timezone) {
     console.log(`[Scheduler] Starting digest run for ${timezone}`);

     const accounts = await getActiveAccounts();
     console.log(`[Scheduler] Processing ${accounts.length} accounts`);

     for (const account of accounts) {
       try {
         // Generate digest using reporter
         const digest = await generateDigest(account.id);

         // Store digest metadata in Supabase before posting
         const { error: storeError } = await supabase
           .from('digest_history')
           .insert({
             account_id: account.id,
             timezone: timezone,
             signals_count: digest.signals?.length || 0,
             trends_count: digest.trends?.length || 0,
             generated_at: new Date().toISOString()
           });

         if (storeError) {
           console.warn(`[Scheduler] Failed to store digest metadata: ${storeError.message}`);
         }

         // Post to Slack (async, non-blocking)
         sendDigest(config.slack.channelId, account.id, digest.blocks)
           .then(result => {
             if (!result.success) {
               console.error(`[Scheduler] Slack post failed for ${account.id}: ${result.error}`);
             }
           })
           .catch(err => {
             console.error(`[Scheduler] Slack post error for ${account.id}: ${err.message}`);
           });

       } catch (error) {
         console.error(`[Scheduler] Failed to generate digest for ${account.id}: ${error.message}`);
         // Continue with next account
       }
     }

     console.log(`[Scheduler] Completed digest run for ${timezone}`);
   }

   function scheduleDigests() {
     // Check every 5 minutes if it's time to send
     // This approach handles DST correctly via luxon
     schedule.scheduleJob('*/5 * * * *', async () => {
       for (const { name, timezone, hour, minute } of config.schedules.digests) {
         const now = DateTime.now().setZone(timezone);
         const runKey = `${timezone}-${now.toISODate()}`;

         // Check if within 5-minute window of target time and hasn't run today
         if (now.hour === hour && now.minute < 5 && !lastRunByTimezone[runKey]) {
           console.log(`[Scheduler] Triggering ${name} digest at ${now.toISO()}`);
           lastRunByTimezone[runKey] = true;

           try {
             await runDigestForTimezone(timezone);
           } catch (error) {
             console.error(`[Scheduler] Error in ${name} run: ${error.message}`);
           }
         }
       }
     });

     console.log('[Scheduler] Digest scheduling started');
     console.log('[Scheduler] Configured timezones:', config.schedules.digests.map(d => `${d.name} (${d.timezone} at ${d.hour}:${String(d.minute).padStart(2, '0')})`).join(', '));
   }

   async function triggerDigestNow(timezone = 'UTC') {
     // Manual trigger for testing
     console.log(`[Scheduler] Manual trigger for ${timezone}`);
     return runDigestForTimezone(timezone);
   }

   module.exports = { scheduleDigests, triggerDigestNow, runDigestForTimezone };
   ```
