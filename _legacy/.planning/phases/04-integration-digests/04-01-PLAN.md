   ```javascript
   const { createClient } = require('@supabase/supabase-js');
   const { DateTime } = require('luxon');

   const supabase = createClient(
     process.env.SUPABASE_URL,
     process.env.SUPABASE_SERVICE_KEY
   );

   async function synthesizeFindings(accountId) {
     const now = DateTime.now();
     const oneDayAgo = now.minus({ days: 1 }).toISO();
     const sevenDaysAgo = now.minus({ days: 7 }).toISO();

     // Fetch signals from researcher agent outputs
     const { data: signals, error: signalsError } = await supabase
       .from('agent_outputs')
       .select('*')
       .eq('account_id', accountId)
       .eq('output_type', 'signal')
       .gte('created_at', oneDayAgo)
       .order('created_at', { ascending: false });

     // Fetch trends from resourcer agent outputs
     const { data: trends, error: trendsError } = await supabase
       .from('agent_outputs')
       .select('*')
       .eq('account_id', accountId)
       .eq('output_type', 'trend')
       .gte('created_at', sevenDaysAgo)
       .order('created_at', { ascending: false });

     // Fetch account details
     const { data: account, error: accountError } = await supabase
       .from('companies')
       .select('*')
       .eq('id', accountId)
       .single();

     if (signalsError || trendsError || accountError) {
       throw new Error(`Supabase error: ${signalsError?.message || trendsError?.message || accountError?.message}`);
     }

     return {
       signals: signals || [],
       trends: trends || [],
       account: account || { id: accountId, name: 'Unknown' }
     };
   }

   module.exports = { synthesizeFindings };
   ```

1. Implement rankSignals() and rankTrends() in src/reporters/ranker.js:
   ```javascript
   const { DateTime } = require('luxon');

   // Configurable scoring weights
   const SCORING_CONFIG = {
     freshness: {
       day: 10,    // < 24h
       week: 5,    // < 7d
       older: 1    // > 7d
     },
     source: {
       official: 5,  // company website, press release
       news: 3,      // techcrunch, crn, etc
       social: 1     // linkedin, twitter
     }
   };

   function scoreByFreshness(createdAt) {
     const age = DateTime.now().diff(DateTime.fromISO(createdAt), 'days').days;
     if (age < 1) return SCORING_CONFIG.freshness.day;
     if (age < 7) return SCORING_CONFIG.freshness.week;
     return SCORING_CONFIG.freshness.older;
   }

   function scoreBySource(source) {
     const officialSources = ['company_website', 'press_release', 'sec_filing'];
     const newsSources = ['techcrunch', 'crn', 'hackernews', 'news'];
     if (officialSources.includes(source?.toLowerCase())) return SCORING_CONFIG.source.official;
     if (newsSources.some(s => source?.toLowerCase()?.includes(s))) return SCORING_CONFIG.source.news;
     return SCORING_CONFIG.source.social;
   }

   function deduplicateBySimilarity(items, threshold = 0.8) {
     // Simple dedup: exact text match (can enhance with Levenshtein later)
     const seen = new Set();
     return items.filter(item => {
       const key = item.text?.toLowerCase()?.trim();
       if (seen.has(key)) return false;
       seen.add(key);
       return true;
     });
   }

   function rankSignals(signals, limit = 3) {
     const deduped = deduplicateBySimilarity(signals);
     const scored = deduped.map(signal => ({
       ...signal,
       score: scoreByFreshness(signal.created_at) + scoreBySource(signal.source)
     }));
     return scored.sort((a, b) => b.score - a.score).slice(0, limit);
   }

   function rankTrends(trends, limit = 3) {
     const deduped = deduplicateBySimilarity(trends);
     const scored = deduped.map(trend => ({
       ...trend,
       score: scoreByFreshness(trend.created_at) + scoreBySource(trend.source)
     }));
     return scored.sort((a, b) => b.score - a.score).slice(0, limit);
   }

   module.exports = { rankSignals, rankTrends, SCORING_CONFIG };
   ```

2. Implement formatter.js with Nunjucks:
   ```javascript
   const nunjucks = require('nunjucks');
   const path = require('path');

   // Configure Nunjucks
   const templatesDir = path.join(__dirname, '../templates');
   nunjucks.configure(templatesDir, { autoescape: true });

   function formatAsBlockKit(account, signals, trends, relevanceCopy = []) {
     const blocks = [
       {
         type: 'header',
         text: { type: 'plain_text', text: `Daily Digest: ${account.name}`, emoji: true }
       },
       {
         type: 'section',
         text: {
           type: 'mrkdwn',
           text: signals.length > 0
             ? `*3 Signals:*\n${signals.map(s => `- ${s.title || 'Signal'}: ${s.text}`).join('\n')}`
             : '*Signals:* No new signals today.'
         }
       },
       {
         type: 'section',
         text: {
           type: 'mrkdwn',
           text: trends.length > 0
             ? `*3 Trends:*\n${trends.map(t => `- ${t.category || 'Trend'}: ${t.text}`).join('\n')}`
             : '*Trends:* No new trends this week.'
         }
       },
       {
         type: 'section',
         text: {
           type: 'mrkdwn',
           text: relevanceCopy.length > 0
             ? `*Why it matters:*\n${relevanceCopy.join('\n')}`
             : '*Why it matters:* Review signals above for outreach opportunities.'
         }
       },
       {
         type: 'actions',
         elements: [
           {
             type: 'button',
             text: { type: 'plain_text', text: 'Useful', emoji: true },
             action_id: `feedback_useful_${account.id}`,
             value: 'useful'
           },
           {
             type: 'button',
             text: { type: 'plain_text', text: 'Not useful', emoji: true },
             action_id: `feedback_notuseful_${account.id}`,
             value: 'not_useful'
           }
         ]
       }
     ];
     return blocks;
   }

   function formatAsNote(account, signals, trends, relevanceCopy = []) {
     return nunjucks.render('attio-note.njk', {
       account,
       signals,
       trends,
       relevanceCopy,
       generatedAt: new Date().toISOString()
     });
   }

   module.exports = { formatAsBlockKit, formatAsNote };
   ```
